            .addScript("classpath:com/bank/config/sql/test-data.sql")
            .build();
    }

    @Profile("production")
    public DataSource jndiDataSource() throws Exception {
        Context ctx = new InitialContext();
        return (DataSource) ctx.lookup("java:comp/env/jdbc/datasource");
